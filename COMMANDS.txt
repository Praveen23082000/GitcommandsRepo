git init = for initialiazing git repository

git status = for listing all untracked files

git add -A = for adding  all untracked files to staging area

git commit -m "connect message"[commit added file to local repository]

git clone = exact copy from remote repository to local workspace

git push = local repository changes to remote repository

git pull = changes from remote repository to local workspace






git config --list = configuration

git config global user name "username" = to add user name

git config global user email "user email" = to add user email
